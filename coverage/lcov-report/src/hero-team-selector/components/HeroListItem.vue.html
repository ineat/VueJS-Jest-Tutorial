      </div>
    </div>

<script lang="ts">
import { Component, Vue, Prop } from "vue-property-decorator";
 
@Component({})
export default class HeroListItem extends Vue {
  @Prop() item: any;
  @Prop() title!: String;
 
  select() {
    this.$store.dispatch("addMember", { type: this.title, member: this.item });
  }
}
</script>
